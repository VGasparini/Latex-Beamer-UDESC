\section{Área}
\begin{frame}
\frametitle{Área: Sistema de Tipos}
    \begin{itemize}
        \item Sistemas de Tipos são uma disciplina de métodos formais (engenharia de software), que
            buscam garantir certas propriedades de programas.
        \item Os desafios da área são: propor sistemas de tipos, programação genérica (generalização de dados e funções),
            garantia de propriedades (verificação), criar algoritmos de inferência e provar propriedades de sistemas de tipos...
    \end{itemize}
\end{frame}

\section{Tema}
\begin{frame}
\frametitle{Tema: Provar propriedades de sistemas de tipos}
    \begin{itemize}
        \item Provar propriedades de sistemas de tipos: provar \textit{soundness} e \textit{completeness}.
    \end{itemize}
\end{frame}

\section{Objetivo e escopo}
\begin{frame}
\frametitle{Objetivo e escopo}
    \begin{itemize}
        \item Algoritmo de inferência para recursão polimórfica criado pelo prof. Cristiano Vasconcellos e prof. Carlos Camarão.
        \item A inferência da recursão polimórfica é provado equivalente ao problema da semi-unificação.
        \item O problema da semi-unificação é indecidível.
        \item Provar que o algoritmo proposto é \textit{completeness}, no entanto a limitação da sintaxe da linguagem impede a criação
            de um caso que entre em \textit{loop} (hipótese).
        \item Provar que o algoritmo proposto não é \textit{completeness} e encontrar um caso que entre em \textit{loop}.
    \end{itemize}
\end{frame}

\section{Método}
\begin{frame}
\frametitle{Método: Prova matemática em Coq}
    \begin{itemize}
        \item Assistentes de provas são programas que verificam a consistência de uma prova.
        \item Assistente Coq.
        \item Prova matemática de \textit{completeness} (ou não) em Coq.
    \end{itemize}
\end{frame}
