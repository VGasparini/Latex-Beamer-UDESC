\section{Introdução}
\begin{frame}
\frametitle{Sistema Milner-Mycroft}
\begin{itemize}
    \item A recursão polimórfica em linguagens de programação foi proposta por \parencite{my:84}
como uma extensão do sistema Damas-Milner \parencite{da:82}.
    \item A modificação permite que o identificador da definição tenha um tipo polimórfico $\sigma$.
    \item Foi mostrado que a inferência no sistema Milner-Mycroft é redutível (em tempo polinomial) ao problema da
semi-unificação, de maneira independente, por \parencite{he:88} e \parencite{kf:88}
\end{itemize}

\section{Semi-unificação}
\begin{frame}
\frametitle{Semi-unificação}
    \begin{itemize}
        \item A semi-unificação é uma generalização da unificação:
            uma substituição $S$ que satisfaça a sequência de inequações
            \begin{align*}
                S(\tau^{1}) \leq S(\tau^{2}) , ..., S(\tau^{n-1}) \leq S(\tau^{n}),
            \end{align*}
            onde $\leq$ é a relação de instância entre tipos: $\tau' \leq \tau$ se, e somente se, existe uma substituição
            $S'$ tal que $S'(\tau') = \tau$.
        \item O caso geral do problema foi mostrado indecidível \parencite{kf:90}.
        \item Isso implica que não há algoritmo para decidir a tipabilidade (\textit{typability}) de um
termo no sistema Milner-Mycroft, ou seja, o sistema também é indecidível.
        \item Alguns casos decidíveis da semi-unificação foram encontrados: para uma inequação
\parencite{ka:88}, para duas variáveis \parencite{le:90} e semi-unificação acíclica \parencite{kf:94}.
    \end{itemize}
\end{frame}

\section{Outros Sistemas com R.P.}
\begin{frame}
\frametitle{Outros Sistemas com Recursão Polimórfica I}
    \begin{itemize}
        \item Tentou-se extensões da linguagem ML que permitem recursão polimórfica, porém \parencite{kf:93} demonstrou que a inferência nessas também é
            indecidível.
        \item A primeira forma de recursão polimórfica decidível foi proposta por \parencite{he:89}, que utilizaria o fato da semi-unificação com uma única inequação ser decidível para
            representar recursões polimórficas com apenas uma chamada recursiva e não aninhadas.
        \item O primeiro sistema de tipos a permitir recursão polimórfica, na realidade, é o Sistema-F \parencite{gi:72}. Em \parencite{ti:90} mostra-se que é possível simular a recursão
            polimórfica com o uso de tipos Rank 3 ou superior.
    \end{itemize}
\end{frame}

\begin{frame}
\frametitle{Outros Sistemas com Recursão Polimórfica II}
    \begin{itemize}
        \item \parencite{ji:95} apresenta uma combinação de Rank 2 com \textit{intersection types}, que é capaz de tipar algumas expressões recursivas e é decidível.
        \item Uma outra abordagem com Rank foi proposta por meio de \textit{Principal Typings}, por \parencite{ji:96}, capaz de tipar alguns casos de recursão polimórfica.
    \end{itemize}
\end{frame}

\begin{frame}
\frametitle{Outros Sistemas com Recursão Polimórfica III}
    \begin{itemize}
                \item \parencite{da:07} propõem, com base no sistema com Rank 2 \textit{intersection type}, o sistema $\vdash^{P}_2$ capaz de tipar, de maneira decidível,
              exemplos mais interessantes de recursão polimórfica.
        \item  Em \parencite{co:08} apresenta-se a relação direta entre recursão polimórfica em sistemas de tipos e em \textit{Abstract Interpretation}, o que resultou num sistema de tipos com
recursão polimórfica decidível.
        \item \parencite{da:11} apresenta a recursão bimórfica, que permite associar um tipo para a chamada recursiva e outro para as chamadas externas da função. Além disso, 
            é possível utilizar chamadas recursivas aninhadas. Mostrou-se que a inferência de tipos nesse sistema é decidível.
    \end{itemize}
\end{frame}

\section{Semi-algoritmos}
\begin{frame}
    \frametitle{Semi-algoritmo \textit{W+}}
    \begin{itemize}
        \item Em \parencite{he:93} é apresentado um semi-algoritmo para encontrar o semiunificador mais geral de um sistema de equações e inequações (SEI).
        \item Na prática programas têm \textit{small types} e esse casos são tratáveis:
            \\
            Teorema 1:
            Tipabilidade com \textit{small types} no sistema Milner-Mycroft é decidível em tempo
            polinomial. Prova em \parencite{he:93}.
        \item O semi-algoritmo para resolver SEI de \parencite{he:93} foi utilizado na construção do semi-algoritmo \textit{W+} 
            em \parencite{em:99}, o qual foi provado ser consistente (\textit{Soundness}) e ter uma 
            completude fraca (\textit{weak completeness}) com as regras do sistema Milner-Mycroft
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Semi-algoritmo \textit{MMo}}
    \begin{itemize}
        \item Foi proposto por \parencite{ca:03} um (semi-)algoritmo chamado \textit{MMo}, o qual foi testado numa boa gama de programas e nunca foi encontrado um
            caso que entra em \textit{loop} ou que rejeita um programa correto.
        \item Apensar dos testes, o grau de incompletude do \textit{MMo} não é conhecido e não se sabe o conjunto de casos
            que consegue tipar.
    \end{itemize}
\end{frame}
